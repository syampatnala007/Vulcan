import React, { useState } from 'react';
import fileSelector from 'file-selector';

const FileExplorer = () => {
  const [fileList, setFileList] = useState([]);

  const handleFileInputChange = async (event) => {
    const files = event.target.files;

    // Use file-selector to get file entries for all selected files and directories
    const fileEntries = await fileSelector(event.target, { readAs: 'dataURL' });

    // Helper function to recursively read directory contents
    const readDirectory = async (directory) => {
      const entries = await directory.createReader().readEntries();
      const files = await Promise.all(
        entries.map(async (entry) => {
          if (entry.isDirectory) {
            return {
              name: entry.name,
              isFile: false,
              content: await readDirectory(entry),
            };
          } else {
            return {
              name: entry.name,
              isFile: true,
              size: entry.size,
              content: fileEntries.find((fileEntry) => fileEntry.name === entry.name)?.content || null,
            };
          }
        })
      );

      return files;
    };

    // Start reading the selected directory
    const filesWithDetails = await Promise.all(
      files.map(async (file) => {
        if (file.isDirectory) {
          return {
            name: file.name,
            isFile: false,
            content: await readDirectory(file),
          };
        } else {
          return {
            name: file.name,
            isFile: true,
            size: file.size,
            content: file.content || null,
          };
        }
      })
    );

    setFileList(filesWithDetails);
  };

  const toggleFolderVisibility = (folderName) => {
    setFileList((prevFileList) =>
      prevFileList.map((file) => {
        if (!file.isFile && file.name === folderName) {
          return { ...file, isFolderVisible: !file.isFolderVisible };
        } else {
          return file;
        }
      })
    );
  };

  const renderFileTree = (fileList) => {
    return fileList.map((file, index) => {
      const isFolder = !file.isFile;
      const isFolderVisible = file.isFolderVisible;

      return (
        <li key={index}>
          {isFolder ? (
            <span onClick={() => toggleFolderVisibility(file.name)}>
              <strong>{file.name}</strong>
            </span>
          ) : (
            <span>
              {file.name} ({file.size} bytes)
            </span>
          )}
          {isFolder && isFolderVisible && <ul>{renderFileTree(file.content)}</ul>}
        </li>
      );
    });
  };

  return (
    <div>
      <input type="file" directory="" webkitdirectory="" onChange={handleFileInputChange} />
      <div>
        <h2>Files and folders in the selected directory:</h2>
        <ul>{renderFileTree(fileList)}</ul>
      </div>
    </div>
  );
};

export default FileExplorer;
